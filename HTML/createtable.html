4. Consultas

Selects Formato de texto

Algunos ejemplos de resultado de consultas. Todas las consulta están en formato texto más arriba.

select nombre as "Nombre Cerveza" from cerveza;

select nombre as "Nombre Cerveza", alcohol as "Graduación" from cerveza order by alcohol desc limit 3;

select nombre as "Nombre Cerveza", alcohol as "Graduación" from cerveza order by alcohol limit 3;

select nombre, estilo from cerveza where estilo ilike 'ale';

select nombre as "Empleados Organizadores" from empleado where tipoEmpleado = 2;

select nombre as "Empleados sin cargo" from empleado where tipoEmpleado not in (1,2);

select nombre as "Empleados Sommeliers" from empleado where tipoEmpleado = 1;

select nombre, cif, telefono from distribuidor order by nombre;

select estilo, count(id) as "Cervezas por Estilo" from cerveza group by estilo;

select e.nombre, count(v.idCerveza) as "Total de Cervezas valoradas" from empleado e join valoracion v on (v.dniEmpleado=e.dni) group by e.nombre order by "Total de Cervezas valoradas";


select c.nombre as "Nombre cerveza", v.puntuacion as Valoración from valoracion v join cerveza c on (v.idCerveza = c.id) order by v.puntuacion desc;
